procedure OnLeftClick (aStr : String);
var
   Str : String;
   Race : Integer;
   CompleteQuest : Integer;
   CurrentQuest : Integer;
begin
   Str := callfunc ('getsenderrace');
   Race := StrToInt (Str);
   if Race <> 1 then begin
      exit;
   end;

   Str := callfunc ('getsendercompletequest');
   CompleteQuest := StrToInt (Str);

   Str := callfunc ('getsendercurrentquest');
   CurrentQuest := StrToInt (Str);
   

   //家券何利阑 父靛绰 过阑 拱阑 荐 乐绊 父甸 荐 乐促.
   if CompleteQuest = 1100 then begin
      Str := 'showwindow .\help\帝王石谷阴阳师1.txt 1';
      print (Str);
      exit;
   end;

   //亲付何利阑 父靛绰 过阑 拱阑 荐 乐绊 父甸绢临 荐 乐促.
   if CurrentQuest = 1450 then begin
      Str := 'showwindow .\help\帝王石谷阴阳师5.txt 1';
      print (Str);
      exit;
   end;
   
   Str := 'showwindow .\help\帝王石谷阴阳师.txt 1';
   print (Str);
   exit;
end;
